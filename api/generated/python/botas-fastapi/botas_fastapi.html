331    async def _handle_activity_async(self, context: TurnContext) -> Optional[InvokeResponse]:
332        if context.activity.type == "invoke":
333            return await self._dispatch_invoke_async(context)
334        handler = self.on_activity or self._handlers.get(context.activity.type.lower())
335        if handler is None:
336            return None
337        dispatch_mode = "on_activity" if self.on_activity else "typed"
338        with _span(
339            "botas.handler",
340            **{"handler.type": context.activity.type, "handler.dispatch": dispatch_mode},
341        ):
342            try:
343                await handler(context)
344            except Exception as exc:
345                metrics = get_metrics()
346                if metrics:
347                    metrics.handler_errors.add(1, {"activity.type": context.activity.type})
